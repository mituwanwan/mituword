## 1. 完整 Schema

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mysql"
  url      = env("DATABASE_URL")
}

// ==================== 用户系统 ====================

model Account {
  id                String  @id @default(cuid())
  userId            String
  type              String
  provider          String
  providerAccountId String
  refresh_token     String? @db.Text
  access_token      String? @db.Text
  expires_at        Int?
  token_type        String?
  scope             String?
  id_token          String? @db.Text
  session_state     String?

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([provider, providerAccountId])
  @@index([userId])
}

model Session {
  id           String   @id @default(cuid())
  sessionToken String   @unique
  userId       String
  expires      DateTime
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
}

model User {
  id            String    @id @default(cuid())
  name          String?
  email         String    @unique
  emailVerified DateTime?
  image         String?
  role          UserRole  @default(USER)
  accounts      Account[]
  sessions      Session[]
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
}

model VerificationToken {
  identifier String
  token      String   @unique
  expires    DateTime

  @@unique([identifier, token])
}

enum UserRole {
  USER
  ADMIN
  EDITOR   // 预留
}

// ==================== 个人资料 ====================

model Profile {
  id          Int      @id @default(autoincrement())
  fullName    String
  avatar      String?
  bio         String?  @db.Text
  email       String?
  githubUrl   String?
  blogUrl     String?
  location    String?
  skills      String?  @db.Text // JSON 数组
  experience  String?  @db.Text // JSON 数组
  education   String?  @db.Text // JSON 数组
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

// ==================== 项目 ====================

model Project {
  id          Int           @id @default(autoincrement())
  slug        String        @unique
  title       String
  description String?       @db.Text
  content     String?       @db.Text // Markdown
  thumbnail   String?
  status      ProjectStatus @default(PLANNING)
  category    String?
  tags        String?       // JSON 数组
  githubUrl   String?
  demoUrl     String?
  featured    Boolean       @default(false)
  order       Int           @default(0)
  views       Int           @default(0)
  createdAt   DateTime      @default(now())
  updatedAt   DateTime      @updatedAt

  @@index([status])
  @@index([featured])
  @@index([category])
}

enum ProjectStatus {
  PLANNING
  IN_PROGRESS
  COMPLETED
  ARCHIVED
}

// ==================== 日记 ====================

model Diary {
  id        Int      @id @default(autoincrement())
  slug      String   @unique
  title     String
  content   String   @db.Text // Markdown
  excerpt   String?  @db.Text
  cover     String?
  tags      String?  // JSON 数组
  published Boolean  @default(true)
  views     Int      @default(0)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@index([published])
  @@index([createdAt])
}

// ==================== 音乐缓存 ====================

model MusicTrack {
  id         Int      @id @default(autoincrement())
  neteaseId  String   @unique // 网易云歌曲ID
  name       String
  artist     String
  album      String?
  coverUrl   String?
  duration   Int?     // 毫秒
  url        String?  // 音频直链（临时，会过期）
  urlExpiry  DateTime? // 链接过期时间
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt
}

model MusicPlaylist {
  id          Int      @id @default(autoincrement())
  neteaseId   String   @unique
  name        String
  description String?
  coverUrl    String?
  trackCount  Int      @default(0)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

// ==================== 好站分享 ⭐ ====================

model SiteShare {
  id          Int      @id @default(autoincrement())
  title       String
  url         String
  description String   @db.Text
  iconUrl     String?
  category    SiteCategory @default(OTHER)
  tags        String?  // JSON 数组
  rating      Int?     // 1-5
  featured    Boolean  @default(false)
  clickCount  Int      @default(0)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  @@index([category])
  @@index([featured])
}

enum SiteCategory {
  DEV_TOOLS    // 开发工具
  DESIGN       // 设计
  LEARNING     // 学习平台
  MUSIC        // 音乐
  GAME         // 游戏
  PRODUCTIVITY // 效率工具
  OTHER        // 其他
}

// ==================== 每日日报 ⭐ ====================

model DailyReport {
  id          Int      @id @default(autoincrement())
  date        DateTime @unique @db.Date
  title       String
  status      ReportStatus @default(DRAFT)
  
  // 内容
  quote       String?  @db.Text  // 今日一句
  quoteAuthor String?           // 句子出处
  
  // 关联推荐（可选）
  trackId     Int?     // 推荐音乐
  siteId      Int?     // 推荐好站
  diaryId     Int?     // 推荐日记
  projectId   Int?     // 推荐项目
  
  // 自定义
  customText  String?  @db.Text // 手动补充内容
  
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

enum ReportStatus {
  DRAFT     // 草稿（自动生成）
  PUBLISHED // 已发布
  HIDDEN    // 隐藏
}
```

## 3. 索引设计说明

| 表 | 索引字段 | 类型 | 用途 |
|----|---------|------|------|
| User | email | Unique | 登录查询 |
| Project | slug | Unique | 详情页路由 |
| Project | status + featured | Composite | 首页精选查询 |
| Diary | slug | Unique | 详情页路由 |
| Diary | published + createdAt | Composite | 列表页查询 |
| SiteShare | category | Normal | 分类筛选 |
| MusicTrack | neteaseId | Unique | 避免重复缓存 |
| DailyReport | date | Unique | 每天一篇 |
